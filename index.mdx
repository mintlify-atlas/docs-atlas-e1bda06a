  <Steps>
    <Step title="Install Vue">
      Create a new Vue project using the official scaffolding tool:
      
      <CodeGroup>
      ```bash npm
      npm create vue@latest
      ```
      ```bash yarn
      yarn create vue
      ```
      ```bash pnpm
      pnpm create vue
      ```
      </CodeGroup>
    </Step>

    <Step title="Create your first component">
      Build reactive interfaces with Vue's intuitive component syntax:

      ```vue App.vue
      <script setup>
      import { ref } from 'vue'

      const count = ref(0)
      </script>

      <template>
        <button @click="count++">
          Count is: {{ count }}
        </button>
      </template>
      ```
    </Step>

    <Step title="Start the dev server">
      Run your application and see live changes:

      ```bash
      npm run dev
      ```

      <Note>
        Your app will be available at `http://localhost:5173`
      </Note>
    </Step>
  </Steps>
